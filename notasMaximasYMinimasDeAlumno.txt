%Maximo
%Producto cartesiano
prod_cart(X, Y):-num(_, X), num(_, Y).

%Seleccion
seleccion(X, Y):-prod_cart(X, Y), X<Y.

%Proyeccion
proyeccion(X):-seleccion(X, _).

%Maximo->NÃºmero que no se encuentre en proyeccion.
max(Alumno, X):-num(Alumno, X), \+ proyeccion(X).

%Minimo
seleccion2(X, Y):-prod_cart(X, Y), X>Y.
proyeccion2(X):-seleccion2(X, _).
min(Alumno, X):-num(Alumno, X), \+ proyeccion2(X).
